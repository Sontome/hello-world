function getPhoneByEmail(email) {
  const data = sheetsdt.getRange(1, 1, sheetsdt.getLastRow(), 2).getValues();
  email = email.trim().toLowerCase();
  for (let i = 0; i < data.length; i++) {
    if (data[i][0] && data[i][0].toString().trim().toLowerCase() === email) {
      return "- " + String(data[i][1] || "");
    }
  }
  return "";
}

function checkPNRready() {
  const lastRow = sheetPNR.getLastRow();
  if (lastRow < 2) return {};
  const data = sheetPNR.getRange(2, 1, lastRow - 1, 10).getValues();

  const grouped = {};
  data.forEach(row => {
    if (row[5] && !row[9]) {
      const key = row[5];
      if (!grouped[key]) grouped[key] = [];
      grouped[key].push(row);
    }
  });
  // Logger.log(grouped)
  return grouped;
}

function filterCompleteGroups(grouped) {
  const result = [];
  for (const key in grouped) {
    const rows = grouped[key];
    const allComplete = rows.every(row => row.slice(0, 8).every(val => val));
    if (allComplete) {
      const colCDE = rows[0].slice(2, 5).concat(rows[0][8]);
      const trimmedRows = rows.map(r => [r[0], r[1], r[6], r[7]]);
      result.push({ idgop: key, info: colCDE, rows: trimmedRows });
    }
  }
  return result;
}

function check(row) {
  const pnr = row[1], hang = row[2];
  const data = sheetlogmail.getDataRange().getValues().slice(1);
  let filtered = data.filter(r => r[4] === pnr && r[6] === "Xuất Vé");
  if (!filtered.length) return [];

  filtered.sort((a, b) => new Date(b[1]) - new Date(a[1]));
  let seen = new Set();
  let unique = filtered.filter(r => {
    if (seen.has(r[7])) return false;
    seen.add(r[7]);
    return true;
  });

  let result = unique.map(r => ({ idMail: r[0], hang,pnr, tripInfo: r.slice(8, 20) }));
  
  return result;
}

function processRows(rows) {
  let result = [];
  for (const row of rows) {
    const r = check(row);
    if (!r.length) return [];
    result = result.concat(r);
  }
  return result;
}

function guiEmailXuatVe() {
  const props = PropertiesService.getScriptProperties();

  if (props.getProperty("guiEmailXuatVe_isRunning") === "true") return;
  props.setProperty("guiEmailXuatVe_isRunning", "true");

  try {
    Logger.log("=== BẮT ĐẦU CHẠY HÀM guiEmailXuatVe ===");

    const a = checkPNRready();
    Logger.log("checkPNRready():");
    Logger.log(JSON.stringify(a, null, 2));

    const b = filterCompleteGroups(a);
    Logger.log("filterCompleteGroups():");
    Logger.log(JSON.stringify(b, null, 2));

    if (!b.length) {
      Logger.log("⛔ Không có group nào đủ điều kiện → dừng");
      return;
    }

    let startIndex = Number(props.getProperty("lastIndex") || "0");
    let batchSize = 5;
    let endIndex = Math.min(startIndex + batchSize, b.length);

    Logger.log(`Xử lý từ index ${startIndex} → ${endIndex - 1}/${b.length - 1}`);

    for (let i = startIndex; i < endIndex; i++) {
      const c = b[i];
      Logger.log(`--- Đang xử lý group ${i}: ${JSON.stringify(c)} ---`);

      const typeraw = c.idgop;
      let type = 0;

      if (typeof typeraw === "string") {
        const match = typeraw.match(/^(\d+)_/);
        if (match) type = parseInt(match[1]);
      }

      Logger.log("type detect = " + type);

      const info = c.info,
        row = c.rows,
        indexs = row.map(r => r[0]),
        listpnr = row.map(r => r[1]);

      Logger.log("info:");
      Logger.log(JSON.stringify(info));

      Logger.log("row:");
      Logger.log(JSON.stringify(row));

      Logger.log("index rows cần update:");
      Logger.log(indexs);

      const d = processRows(row);
      Logger.log("Kết quả processRows:");
      Logger.log(JSON.stringify(d, null, 2));

      if (!d.length) {
        Logger.log("⛔ Không tìm thấy dữ liệu Email + PDF → bỏ group này");
        continue;
      }

      Logger.log("Gọi processMultiplePdfFromEmail...");
      const result = processMultiplePdfFromEmail(d, info[1], info[2], info[0], info[3], type);

      Logger.log("Kết quả gửi email:");
      Logger.log(JSON.stringify(result));

      if (result.status === "OK") {
        indexs.forEach(idx => {
          Logger.log(`Update link gửi email vào dòng ${idx}`);
          sheetPNR.getRange(idx, 10).setValue(result.link);
        });

        Logger.log("Push Telegram: " + info[0]);
        pushTele(info[0], listpnr);

        const lastRow = sheetLogPnrEmailKhach.getLastRow();
        let dataToAdd = [];

        const tripMap = {};
        d.forEach(item => {
          if (item.pnr) tripMap[item.pnr] = item.tripInfo || [];
        });

        Logger.log("TripMap:");
        Logger.log(JSON.stringify(tripMap));

        row.forEach(r => {
          const pnr = r[1];
          const tripInfo = tripMap[pnr] || [];

          dataToAdd.push([
            info[2], // họ
            info[1], // tên
            info[0], // email
            getPhoneByEmail(info[0]),
            pnr,
            ...tripInfo,
            Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "dd/MM/yyyy HH:mm")
          ]);
        });

        Logger.log("Ghi log email khách:");
        Logger.log(JSON.stringify(dataToAdd));

        if (dataToAdd.length) {
          sheetLogPnrEmailKhach
            .getRange(lastRow + 1, 1, dataToAdd.length, dataToAdd[0].length)
            .setValues(dataToAdd);
        }
      }
    }

    Logger.log("=== KẾT THÚC HÀM guiEmailXuatVe ===");

  } finally {
    props.deleteProperty("guiEmailXuatVe_isRunning");
  }
}


function capitalizeFullName(name) {
  return name.split(" ").map(w => w.charAt(0).toUpperCase() + w.slice(1)).join(" ");
}
